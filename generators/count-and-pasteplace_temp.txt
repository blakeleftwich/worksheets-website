function placeObjects(svg, { iconId, count, x, y, w, h, answerBox = null, problemIndex = 0, problemCount = 8, perRowMax = 5 }) {
  const problemGroup = svgEl("g", { "data-problem": problemIndex });
  const pad = 10;

  // Available area for icons (inside cell padding)
  let aX = x + pad;
  let aY = y + pad;
  let aW = w - pad * 2;
  let aH = h - pad * 2;

  // Reduce available height to avoid answer box
  if (answerBox) {
    const abTop = answerBox.y - 6; // 6px buffer above answer box
    const maxH = abTop - aY;
    if (maxH < aH) aH = maxH;
  }

  // Fixed row layout based on count range:
  // Items <= perRowMax: single row; items > perRowMax: wrap to 2 rows
  let bestCols, bestRows;
  if (count <= perRowMax) {
    bestCols = count;
    bestRows = 1;
  } else {
    bestCols = perRowMax;
    bestRows = Math.ceil(count / perRowMax);
  }
  const spacingRatio = 0.15;
  const sizeByW = aW / (bestCols + (bestCols - 1) * spacingRatio);
  const sizeByH = aH / (bestRows + (bestRows - 1) * spacingRatio);
  const bestSize = Math.min(sizeByW, sizeByH);

  let iconSize = Math.max(18, bestSize);
  const spacing = iconSize * spacingRatio;
  const gridW = bestCols * iconSize + (bestCols - 1) * spacing;
  const gridH = bestRows * iconSize + (bestRows - 1) * spacing;
  const startX = aX + (aW - gridW) / 2;
  const startY = aY + (aH - gridH) / 2;

  let idx = 0;
  for (let r = 0; r < bestRows && idx < count; r++) {
    for (let c = 0; c < bestCols && idx < count; c++) {
      const ix = startX + c * (iconSize + spacing);
      const iy = startY + r * (iconSize + spacing);
      problemGroup.appendChild(svgEl("use", {
        href: `#${iconId}`,
        x: ix, y: iy,
        width: iconSize, height: iconSize
      }));
      idx++;
    }
  }

  svg.appendChild(problemGroup);
}

  // Calculate size with guaranteed spacing
  const availableWidth = w - (count + 1) * 8; // 8px spacing minimum
  const maxItemWidth = availableWidth / count;
